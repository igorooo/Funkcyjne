class Time(var hour: Int){
  if(hour < 0) hour = 0

  def time: Int = hour

  def time_= (newHour :  Int){
    if(newHour < 0) hour = 0
    else hour = newHour
  }

}

object Time{
  def apply(h :Int = 0) = new Time(h)
}
